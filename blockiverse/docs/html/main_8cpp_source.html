111 void testLogin(bvnet::session *s,KeyPair *ckey,bool *authOk,bool *doneFlag) {
112  std::string erc=s->getarg<std::string>();
113  std::string rcsha;
114  LOCK_COUT
115  std::cout << "Decrypting server challenge..." << std::endl;
116  UNLOCK_COUT
117  std::string rc=RSA::Decrypt(erc,ckey->GetPrivateKey());
118  LOCK_COUT
119  std::cout << " decrypted: " << rc << std::endl;
120  UNLOCK_COUT
121  SHA1 rcdig;
122  rcdig.addBytes(rc.c_str(),rc.size());
123  unsigned char *dig=rcdig.getDigest();
124  std::ostringstream ss;
125  for (int i=0;i<20;++i)
126  ss << std::setfill('0') << std::setw(2) << std::hex
127  << (unsigned int)dig[i];
128  rcsha=ss.str();
129  free(dig);
130  LOCK_COUT
131  std::cout << " SHA1: " << rcsha << std::endl;
132  UNLOCK_COUT
133  s->send_string(rcsha);
134  s->send_call(1 /* serverRoot */,2 /* AnswerChallenge */,
135  boost::bind(loginDone,s,
136  authOk,doneFlag),
137  1 /* expects one argument */);
138 }

311  /*
312  The most important function of the engine is the 'createDevice'
313  function. The Irrlicht Device can be created with it, which is the
314  root object for doing everything with the engine.
315  createDevice() has 7 paramters:
316  deviceType: Type of the device. This can currently be the Null-device,
317  the Software device, DirectX8, DirectX9, or OpenGL. In this example we use
318  EDT_SOFTWARE, but to try out, you might want to change it to
319  EDT_NULL, EDT_DIRECTX8 , EDT_DIRECTX9, or EDT_OPENGL.
320  windowSize: Size of the Window or FullscreenMode to be created. In this
321  example we use 640x480.
322  bits: Amount of bits per pixel when in fullscreen mode. This should
323  be 16 or 32. This parameter is ignored when running in windowed mode.
324  fullscreen: Specifies if we want the device to run in fullscreen mode
325  or not.
326  stencilbuffer: Specifies if we want to use the stencil buffer for drawing shadows.
327  vsync: Specifies if we want to have vsync enabled, this is only useful in fullscreen
328  mode.
329  eventReceiver: An object to receive events. We do not want to use this
330  parameter here, and set it to 0.
331  */
