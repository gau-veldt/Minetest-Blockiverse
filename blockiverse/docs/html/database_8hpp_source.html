209  query_result run(sqlite3_stmt *stmt) {
210  query_result data(new Result);
211  int rc;
212  do {
213  rc=sqlite3_step(stmt);
214  if (rc==SQLITE_ROW) {
215  data->resize(data->size()+1);
216  rowResult cur=data->back();
217  int nrows=sqlite3_column_count(stmt);
218  for (int i=0;i<nrows;++i) {
219  LOCK_COUT
220  cout << "[DB] query " << stmt
221  << ": result row " << i+1 << "/" << nrows
222  << endl;
223  UNLOCK_COUT
224  switch (sqlite3_column_type(stmt,i)) {
225  case SQLITE_NULL:
226  cur[i]=dbValue::ptr(new dbValue);
227  break;
228  case SQLITE_INTEGER:
229  cur[i]=dbValue::ptr(new dbValue((s64)sqlite3_column_int(stmt,i)));
230  break;
231  case SQLITE_FLOAT:
232  cur[i]=dbValue::ptr(new dbValue((double)sqlite3_column_double(stmt,i)));
233  break;
234  case SQLITE_TEXT:
235  cur[i]=dbValue::ptr(new dbValue((const char *)sqlite3_column_text(stmt,i)));
236  break;
237  case SQLITE_BLOB:
238  cur[i]=dbValue::ptr(
239  new dbValue(
240  (const char *)sqlite3_column_text(stmt,i),
241  sqlite3_column_bytes(stmt,i)
242  ));
243  break;
244  }
245  }
246  }
247  } while (rc==SQLITE_ROW);
248  if (rc!=SQLITE_DONE) {
249  DBError::busy_aware_throw(rc,sqlite3_errmsg(db));
250  }
251  LOCK_COUT
252  cout << "[DB] " << stmt << ": " << data->size() << " rows returned." << endl;
253  UNLOCK_COUT
254  return data;
255  }
