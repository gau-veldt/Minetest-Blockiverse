374  inline int registry::register_object(object *ob) {
375  int chosen_slot; // local value other threads can't modify
385  if (ob==NULL) {
386  /* refuse to register a NULL pointer */
387  throw object_null();
388  }
389 
390  {
391  scoped_lock lock(*synchro);
392 
393  if (objects.size()>=reg_objects_softmax) {
394  /*
395  ** enforce an object store softmax
396  **
397  ** A softmax exception can be caught and the offending session
398  ** simply disconnected whether due to a configuration/program
399  ** error (softmax too low), or a deliberate attack by a malicious
400  ** client attempting to create enough objects to exhaust the free
401  ** store hardmax which may lead to a system crash and subsequent
402  ** collateral damage.
403  */
404  throw registry_full();
405  }
406 
407  /*
408  For security do not assume next_slot+1
409  is an unoccupied slot.
410 
411  An attacker could use rapid creation and deletion of
412  objects to execute a wraparound attack on next_slot
413  then register a new object in a low numbered slot to
414  overwrite an existing object in the registry.
415  */
416  omap_select_id &table=objects.by<object_id>();
417  omap_iter_byid row=table.find(next_slot);
418  while (row!=table.end()) {
419  ++next_slot;
420  /* skip 0 as it's reserved */
421  if (next_slot==0) ++next_slot;
422  row=table.find(next_slot);
423  }
424  /* invariant: objects[next_slot] empty */
425 
426  /* make thread-safe copy for retval */
427  chosen_slot=next_slot;
428 
429  /* register object */
430  objects.insert(ob_relation(chosen_slot,ob));
431  }
432  return chosen_slot;
433  }

441  inline bool registry::unregister(u32 id) {
442  /*
443  ** remove object from registry
444  ** and notify upstream event sink
445  */
446  scoped_lock lock(*synchro);
447 
448  omap_select_id &table=objects.by<object_id>();
449  omap_iter_byid victim=table.find(id);
450  if (victim!=table.end()) {
451  notify(id);
452  table.erase(victim);
453  return true;
454  }
455  return false;
456  }

742  inline void session::on_recv(const boost::system::error_code &ec,size_t rlen) {
743  _opcode_read_queued=false;
744  if (isActive) {
745  if (!ec) {
746  int ich=int((unsigned char)in_ch);
747  if (_float_or_semi) {
748  if (in_ch==';') {
749  _float_or_semi=false;
750  float flt;
751  std::istringstream cvt(_fpstr);
752  cvt >> flt;
753  argstack.push(flt);
754  check_argnotify();
755  } else {
756  _fpstr+=in_ch;
757  }
758  } else {
759  u32 bsize;
760  switch(in_ch) {
761  /* integer value of 2^N bytes */
762  case '0':
763  case '1':
764  case '2':
765  case '3':
766  /*case '4':
767  case '5':
768  case '6':
769  case '7':
770  case '8':
771  case '9':*/
772  bsize=pow2_tbl[in_ch-'0'];
773  *((s64*)in_s64)=0;
774  boost::asio::async_read(
775  *conn,
776  boost::asio::buffer(in_s64,bsize),
777  boost::bind(&session::on_recv_s64,this,
778  boost::asio::placeholders::error,
779  bsize));
780  break;
781  case '-':
782  _neg_int=true;
783  break;
784  case '+':
785  _neg_int=false;
786  break;
787  case 'f':
788  _float_or_semi=true;
789  _fpstr.clear();
790  break;
791  case '"':
792  case 'b':
793  boost::asio::async_read(
794  *conn,
795  boost::asio::buffer(in_idx,4),
796  boost::bind(&session::on_recv_len,this,
797  boost::asio::placeholders::error,
798  boost::asio::placeholders::bytes_transferred));
799  break;
800  case 'o':
801  boost::asio::async_read(
802  *conn,
803  boost::asio::buffer(in_idx,4),
804  boost::bind(&session::on_recv_oref,this,
805  boost::asio::placeholders::error,
806  boost::asio::placeholders::bytes_transferred));
807  break;
808  case '.':
809  boost::asio::async_read(
810  *conn,
811  boost::asio::buffer(in_idx,4),
812  boost::bind(&session::on_recv_call_obid,this,
813  boost::asio::placeholders::error));
814  break;
815  case '~':
816  boost::asio::async_read(
817  *conn,
818  boost::asio::buffer(in_idx,4),
819  boost::bind(&session::on_recv_dead_obid,this,
820  boost::asio::placeholders::error));
821  break;
822  default:
823  LOCK_COUT
824  std::cout << "session [" << this << "] recv len="
825  << rlen << " 0x"
826  << std::setw(2) << std::setfill('0') << std::hex << ich;
827  if (ich>31 && ich<128)
828  std::cout << ' ' << in_ch;
829  std::cout << std::endl;
830  UNLOCK_COUT
831  }
832  }
833  } else {
834  LOCK_COUT
835  if (_float_or_semi) {
836  std::cout << "session [" << this
837  << "] expected floatnum; got EOF"
838  << " (" << ec << ")"
839  << std::endl;
840 
841  } else {
842  std::cout << "session [" << this
843  << "] expected opcode got EOF"
844  << " (" << ec << ")"
845  << std::endl;
846  }
847  UNLOCK_COUT
848  isActive=false;
849  }
850  }

925  inline void session::encode(const boost::any &raw) {
926  std::ostringstream ss;
927  u32 idx;
928  method_call mc(0,0,NULL,0);
929  const char *idx_byte=(const char*)&idx;
930  type_map::iterator tmi=typeMap.find(raw.type().name());
931  if (tmi==typeMap.end()) {
932  LOCK_COUT
933  std::cout << "<unknown \"" << raw.type().name() << "\">" << std::endl;
934  UNLOCK_COUT
935  } else {
936  s64 val;
937  u64 mag,mag2;
938  int log2,bytes;
939  float flt;
940  switch (tmi->second) {
941  case vtInt:
942  val=boost::any_cast<s64>(raw);
943  mag=(val<0)?0-val:val;
944  if (val<0) {
945  ss << '-';
946  }
947  bytes=1;
948  mag2=mag;
949  while (mag2>255) {
950  ++bytes;
951  mag2>>=8;
952  }
953  log2=log2_tbl[bytes-1];
954  ss << log2;
955  bytes=pow2_tbl[log2];
956  while (bytes>0) {
957  ss << (const char)(mag&0xff);
958  mag>>=8;
959  --bytes;
960  }
961  break;
962  case vtFloat:
963  flt=boost::any_cast<float>(raw);
964  ss << 'f' << flt << ';';
965  break;
966  case vtBlob:
967  case vtString:
968  idx=boost::any_cast<std::string>(raw).size();
969  ss << '"'
970  << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
971  ss << boost::any_cast<std::string>(raw);
972  break;
973  case vtObref:
974  idx=boost::any_cast<obref>(raw).id;
975  ss << 'o'
976  << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
977  break;
978  case vtDeath:
979  idx=boost::any_cast<ob_is_gone>(raw).id;
980  ss << '~'
981  << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
982  break;
983  case vtMethod:
984  mc=boost::any_cast<method_call>(raw);
985  LOCK_COUT
986  std::cout << "call #" << mc.id << '.' << mc.idx
987  << "() cb=" << mc.callbk
988  << ", when stack+=" << mc.rcount
989  << std::endl;
990  UNLOCK_COUT
991  ss << '.';
992  idx=mc.id;
993  ss << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
994  idx=mc.idx;
995  ss << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
996  break;
997  }
998  std::string *dynstr=new std::string(ss.str());
999  if (tmi->second==vtMethod
1000  && mc.callbk) {
1001  // only if callback not null
1002  if (mc.rcount==0) {
1003  // a callback of rcount 0 notifies
1004  // when the queued method call processed
1005  mc.callbk();
1006  } else {
1007  // otherwise the notify is set for when
1008  // argstack reaches size current_size+rcount
1009  argnotify.push(mc);
1010  }
1011  }
1012  boost::asio::async_write(
1013  *conn,
1014  boost::asio::buffer(*dynstr,dynstr->size()),
1015  boost::bind(&session::on_write_done,this,dynstr));
1016  }
1017  }
