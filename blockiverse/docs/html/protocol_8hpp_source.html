519  inline int registry::register_object(object *ob) {
520  int chosen_slot; // local value other threads can't modify
530  if (ob==NULL) {
531  /* refuse to register a NULL pointer */
532  throw object_null();
533  }
534 
535  {
536  scoped_lock lock(*synchro);
537 
538  if (objects.size()>=reg_objects_softmax) {
539  /*
540  ** enforce an object store softmax
541  **
542  ** A softmax exception can be caught and the offending session
543  ** simply disconnected whether due to a configuration/program
544  ** error (softmax too low), or a deliberate attack by a malicious
545  ** client attempting to create enough objects to exhaust the free
546  ** store hardmax which may lead to a system crash and subsequent
547  ** collateral damage.
548  */
549  throw registry_full();
550  }
551 
552  /*
553  For security do not assume next_slot+1
554  is an unoccupied slot.
555 
556  An attacker could use rapid creation and deletion of
557  objects to execute a wraparound attack on next_slot
558  then register a new object in a low numbered slot to
559  overwrite an existing object in the registry.
560  */
561  omap_select_id &table=objects.by<object_id>();
562  omap_iter_byid row=table.find(next_slot);
563  while (row!=table.end()) {
564  ++next_slot;
565  /* skip 0 as it's reserved */
566  if (next_slot==0) ++next_slot;
567  row=table.find(next_slot);
568  }
569  /* invariant: objects[next_slot] empty */
570 
571  /* make thread-safe copy for retval */
572  chosen_slot=next_slot;
573 
574  /* register object */
575  objects.insert(ob_relation(chosen_slot,ob));
576  }
577  return chosen_slot;
578  }

1234  case vtString:

1239  break;

1254  ss << idx_byte[0] << idx_byte[1] << idx_byte[2] << idx_byte[3];
