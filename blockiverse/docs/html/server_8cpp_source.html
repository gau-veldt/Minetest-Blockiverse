204 void serverRoot::dmc_LoginClient(value_queue &vqueue) {
205  /* in: string: client pubkey modulus
206  ** string: client pubkey exponent
207  **
208  ** out: string: challenge (encrpyted w/pubkey)
209  */
210  string sMod,sExp,eChal;
211  sExp=ctx.getarg<string>(); /* LIFO is exponent */
212  cli_pub_exp=BigInt(sExp);
213  sMod=ctx.getarg<string>();
214  cli_pub_mod=BigInt(sMod);
215  clientKey=new Key(cli_pub_mod,cli_pub_exp);
216  for (int i=0;i<8;++i)
217  randbits[i]=entropy();
218  unsigned char *randbyte=(unsigned char *)randbits;
219  challenge.clear();
220  /*
221  ** Dividing down to valid printables (32-127) makes
222  ** the challenge effectively [192..208] bits or 6-6.5
223  ** bits per byte. An SHA1 may potentially collide past
224  ** 160 bits thus making 160 bits the minimum, a lower
225  ** bound ND randomness of 192 bits will suffice.
226  **
227  ** In release builds where the challenge won't be
228  ** displayed dividing down to printables will not
229  ** be necessary and the entire 256 bits may be
230  ** utilized.
231  */
232  for (int i=0;i<31;++i) {
233  challenge+=(unsigned char)(32+((int)(((float)randbyte[i])/2.68421)));
234  }
235  eChal=RSA::Encrypt(challenge,*clientKey);
236  vqueue.push(eChal);
237 }
