59 class serverRoot : public bvnet::object {
60 private:
61  BigInt cli_pub_mod;
62  BigInt cli_pub_exp;
63  Key *clientKey;
64  bool clientValid;
65  std::string challenge;
66  unsigned int randbits[8];
67 public:
68  serverRoot(bvnet::session &sess)
69  : bvnet::object(sess) {
70  clientValid=false;
71  challenge="";
72  clientKey=NULL;
73  }
74  virtual ~serverRoot() {
75  if (clientKey!=NULL)
76  delete clientKey;
77  }
78 
79  virtual const char *getType() {return "serverRoot";}
80  virtual void methodCall(unsigned int method) {
81  bvnet::scoped_lock lock(ctx.getMutex());
82  bvnet::value_queue &vqueue=ctx.getSendQueue();
83  switch(method) {
84  case 0: /* GetType */
85  /* emit object type to output queue as string */
86  vqueue.push(std::string(getType()));
87  break;
88  case 1: /* LoginClient */
89  {
90  /* in: string: client pubkey modulus
91  ** string: client pubkey exponent
92  **
93  ** out: string: challenge (encrpyted w/pubkey)
94  */
95  std::string sMod,sExp,eChal;
96  sExp=ctx.getarg<std::string>(); /* LIFO is exponent */
97  cli_pub_exp=BigInt(sExp);
98  sMod=ctx.getarg<std::string>();
99  cli_pub_mod=BigInt(sMod);
100  clientKey=new Key(cli_pub_mod,cli_pub_exp);
101  for (int i=0;i<8;++i)
102  randbits[i]=entropy();
103  unsigned char *randbyte=(unsigned char *)randbits;
104  challenge.clear();
105  /*
106  ** Dividing down to valid printables (32-127) makes
107  ** the challenge effectively [192..208] bits or 6-6.5
108  ** bits per byte. An SHA1 may potentially collide past
109  ** 160 bits thus making 160 bits the minimum, a lower
110  ** bound ND randomness of 192 bits will suffice.
111  **
112  ** In release builds where the challenge won't be
113  ** displayed dividing down to printables will not
114  ** be necessary and the entire 256 bits may be
115  ** utilized.
116  */
117  for (int i=0;i<31;++i) {
118  challenge+=(unsigned char)(32+((int)(((float)randbyte[i])/2.68421)));
119  }
120  eChal=RSA::Encrypt(challenge,*clientKey);
121  vqueue.push(eChal);
122  }
123  break;
124  case 2: /* AnswerChallenge */
125  {
126  /*
127  ** in: string: SHA1 matching decrypted challenge string
128  ** (string encrypted with client privkey)
129  **
130  ** out: integer: 1 if accepted
131  **
132  ** NB: invalid response drops the connection
133  */
134  s64 authOk=0;
135  std::string hChal,answer;
136  answer=ctx.getarg<std::string>();
137  SHA1 sha;
138  sha.addBytes(challenge.c_str(),challenge.size());
139  unsigned char *dig=sha.getDigest();
140  std::ostringstream ss;
141  for (int i=0;i<20;++i)
142  ss << std::setfill('0') << std::setw(2) << std::hex
143  << (unsigned int)dig[i];
144  hChal=ss.str();
145  free(dig);
146  LOCK_COUT
147  std::cout << "[server] Client answered challenge:" << std::endl
148  << " mine: " << hChal << std::endl
149  << " client: " << answer << std::endl
150  << " same: " << (hChal==answer) << std::endl;
151  UNLOCK_COUT
152  if (hChal==answer) {
153  authOk=1;
154  clientValid=true;
155  }
156  vqueue.push(authOk);
157  if (!clientValid) {
158  ctx.disconnect();
159  }
160  }
161  break;
162  case 3: /* GetAccount */
163  {
164  /*
165  ** Called by authenticated client to
166  ** log on to a user account
167  **
168  ** in: string username
169  ** string password
170  **
171  ** out: objectref account or
172  ** int result code
173  **
174  ** username is the username of the account
175  ** to log onto either the username owned by
176  ** the client or anotehr user who has linked
177  ** (whitelisted) this client
178  **
179  ** as a special case if client does not currently own a
180  ** user account and the specified username is not in use
181  ** by another account it will be created and registered
182  ** with this client's pubkey as its owner
183  **
184  ** a pasword string must always be sent but
185  ** only used when the username being logged
186  ** onto via linked (whitelisted) client
187  **
188  ** nonempty password are sent encrypted
189  ** (by client with privkey) and server
190  ** will decode with connected client's
191  ** pubkey
192  **
193  ** returns either the objectref
194  ** of the user account on success
195  ** or an integer result code on
196  ** failure:
197  **
198  ** 0: invalid client (did not answer chellenge)
199  ** 1: unauthorized (not owner of the account or incorrect
200  ** password supplied for linked (whitelisted) account
201  **
202  ** calling this method from unvalidated
203  ** client causes disconnection after
204  ** returning a result code of 0
205  **
206  */
207  std::string pass,user;
208  pass=ctx.getarg<std::string>(); // LIFO is password
209  user=ctx.getarg<std::string>();
210 
211  if (clientValid) {
212  // TODO: check for new account creation,
213  // logon to own account or logon to
214  // linked (whitelisted) account
215  } else {
216  // invalid (unauthorized) client
217  vqueue.push(s64(0));
218  ctx.disconnect();
219  }
220  }
221  break;
222  }
223  }
224 };
